#Loading in the data

library(haven)
## Warning: package 'haven' was built under R version 4.5.2
data <-read_sav("t821_martanew.sav")
subdata <- data[, c('gender','age','wbccat','anthra_dose','relapse',
                    'death','time_surv',
                    'event','time_event',
                    'event_CR','time_CR')]

#Estimators time since diagnosis

#MLE parameters:
x_vals <- seq(0, 10, length.out = 1000)#X vector
#MLE parameters: for EFS
r_event = sum(subdata$event[!is.na(subdata$time_event)])
T_event = sum(subdata$time_event[!is.na(subdata$time_event)])
lambda_event = r_event/T_event
y_event_vals <- exp(-lambda_event * x_vals)
#MLE parameters: for OS
r_death = sum(subdata$death[!is.na(subdata$time_surv)])
T_death = sum(subdata$time_surv[!is.na(subdata$time_surv)])
lambda_death = r_death/T_death
y_death_vals <- exp(-lambda_death * x_vals)

#Plotting estimators time since diagnosis

#EFS nonparametric and parametric
subdata_event <- subdata[subdata$time_event > 0,c('event','time_event')]
fit_event_km <- survfit(Surv(time_event, event) ~ 1, data = subdata_event)#Kaplan-Meier
fit_event_fh <- survfit(Surv(time_event, event) ~ 1, 
                        data = subdata_event, type = "fh")#Fleming-Harrington

df_event_km <- data.frame(time = fit_event_km$time,
                          surv = fit_event_km$surv,
                          lower = fit_event_km$lower,
                          upper = fit_event_km$upper,
                          Estimator = "Kaplan–Meier")

df_event_fh <- data.frame(time = fit_event_fh$time,
                          surv = fit_event_fh$surv,
                          lower = fit_event_fh$lower,
                          upper = fit_event_fh$upper,
                          Estimator = "Fleming–Harrington")
df_event_l <- data.frame(time = x_vals,
                         surv = y_event_vals,
                         lower = NA,
                         upper = NA,
                         Estimator = "Exponential distribution")

df_event <- rbind(df_event_km, df_event_fh, df_event_l)#combining the dataframes

plot_event <- ggplot(df_event, aes(time, surv, color = Estimator)) +
  geom_step()+
  labs(
    title = "Event Free Survival",
    x = "Time since diagnosis (in years)",
    y = "Survival function"
    )+
  xlim(0,10)+
  ylim(0,1)+
  theme(legend.position = c(0.2, 0.2),
        legend.background = element_rect(fill = "white"))
print(plot_event)
